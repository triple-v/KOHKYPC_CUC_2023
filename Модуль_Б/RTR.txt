en    //  Входим в привилегированный режим



conf t    // Входим в настройки конфигурации

no logging on   // Отключаем вывод сообщений в консоли, мешающих вводу команд



int g1  // Входим в настройки первого интерфейса сетевой карты

ip addr 10.0.137.12 255.255.255.0   // Настройка IP адресации первого интерфейса

ip gateway 10.0.137.1 255.255.255.0   // Настройка шлюза для первого интерфейса

no shutdown   // Включаем запуск интерфейса при последующей загрузке ВМ

exit    // Выходим из настроек первого интерфейса



int g2    // Входим в настройки второго интерфейса сетевой карты

ip addr 192.168.0.1 255.255.255.128   // Настройка IP адресации второго интерфейса

no shutdown   // Включаем запуск интерфейса при последующей загрузке ВМ

exit    // Выходим из настроек второго интерфейса



reload    // Перезагружаем ВМ, на все вопросы в консоли пишем yes и жмём Enter
